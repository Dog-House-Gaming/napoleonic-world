country_event = {
	id = spain.2
	title = spain.2.t
	desc = spain.2.d
	picture = GFX_report_event_generic_riots
	
	fire_only_once = yes
	
	trigger = {
		tag = SPR
		SPR = {
		OR = {
			NOT = {controls_state = 41}
			has_capitulated = yes
			is_subject = yes
			is_puppet = yes
		}
		}
	}
	
	immediate = {
 		hidden_effect = {
 			end_puppet = ARG
			end_puppet = COL
			end_puppet = MEX
			ARG = {
				set_politics = {
					ruling_party = democratic
				}
			}
			COL = {
				set_politics = {
					ruling_party = democratic
				}
			}
			MEX = {
				set_politics = {
					ruling_party = democratic
				}
			}
 		} 		
 	}
	
	option = {
		name = spain.2.a
		
		if = {
			limit = {
				SPR = {
				is_puppet = no
				}
			}
			ARG = {
				leave_faction = yes
				white_peace = FRA
				add_ideas = {
					bra_fighting_for_independence
				}
			}
			COL = {
				leave_faction = yes
				white_peace = FRA
				add_ideas = {
				bra_fighting_for_independence
				}
			}
			MEX = {
				leave_faction = yes
				white_peace = FRA
				add_ideas = {
				bra_fighting_for_independence
				}
			}
			declare_war_on = {
				target = ARG
				type = annex_everything
			}
			declare_war_on = {
				target = COL
				type = annex_everything
			}
			declare_war_on = {
				target = MEX
				type = annex_everything
			}
			
		}
	}
}
